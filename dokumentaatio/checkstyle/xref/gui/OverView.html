6   package gui;
7   
8   import javafx.geometry.Insets;
9   import javafx.geometry.Pos;
10  import javafx.scene.Parent;
11  import javafx.scene.layout.GridPane;
12  import javafx.scene.text.Text;
13  import logic.Budgeter;
14  import logic.Task;
15  import logic.Tasklist;
16  import logic.User;
17  
18  /**
19   * Class creates an overview of tasks and budget info to be displayed in a
20   * GridPane.
21   *
22   */
23  public class OverView {
24  
25      private Tasklist tasklist;
26      private Budgeter budgeter;
27      private User user;
28  
29      /**
30       * Constructor that sets the objects used to display data the user has given
31       * and show calculations based off the data.
32       *
33       *
34       * @param tasklist list of tasks given by the object initializing this
35       * object
36       * @param budgeter budgeting information given by the object initializing
37       * this object
38       * @param user user information given by the object initializing this object
39       *
40       * @see Tasklist
41       * @see Budgeter
42       * @see User
43       */
44      public OverView(Tasklist tasklist, Budgeter budgeter, User user) {
45          this.tasklist = tasklist;
46          this.budgeter = budgeter;
47          this.user = user;
48      }
49  
50      /**
51       * Method returns a GridPane view of task and budget information given by
52       * the user.
53       *
54       * @return GridPane of budget and task information
55       */
56      public Parent getView() {
57          GridPane setting = new GridPane();
58  
59          int y = 1;
60  
61          setting.setAlignment(Pos.CENTER);
62          setting.setVgap(10);
63          setting.setHgap(10);
64          setting.setPadding(new Insets(10, 10, 10, 10));
65          setting.add(new Text("Tehtävät"), 0, 0);
66          setting.add(new Text("Tehtäväajat"), 1, 0);
67          setting.add(new Text("Motiivit"), 2, 0);
68          setting.add(new Text("Tulot"), 4, 0);
69          setting.add(new Text("Menot"), 5, 0);
70          setting.add(new Text("Tulot:"), 7, 0);
71          setting.add(new Text("Menot:"), 7, 2);
72          setting.add(new Text("Budjetin erotus:"), 7, 4);
73          setting.add(new Text(Double.toString(budgeter.totalIncome())), 7, 1);
74          setting.add(new Text(Double.toString(budgeter.totalExpenses())), 7, 3);
75          setting.add(new Text(Double.toString(budgeter.checkBalance())), 7, 5);
76  
77          for (Task t : tasklist.getTasks().values()) {
78  
79              setting.add(new Text(t.getName()), 0, y);
80              setting.add(new Text(Integer.toString(t.getTime())), 1, y);
81              setting.add(new Text(t.getMotive()), 2, y);
82              y++;
83              if (y == 15) {
84                  setting.add(new Text("..."), 0, y);
85                  setting.add(new Text("..."), 1, y);
86                  y++;
87                  break;
88              }
89          }
90          y++;
91          setting.add(new Text("Kokonaisaika"), 0, y);
92          setting.add(new Text(Integer.toString(tasklist.hoursRequired()) + " tuntia & " + Integer.toString(tasklist.minutesRequired()) + " minuuttia"), 1, y);
93          y++;
94          setting.add(new Text("Tehtävien suorittamiseen kuluvat päivät"), 0, y);
95          setting.add(new Text(Integer.toString(tasklist.daysRequired(user.getDailyCapacity()))), 1, y);
96          y = 1;
97  
98          for (String n : budgeter.getIncome().keySet()) {
99              setting.add(new Text(n), 4, y);
100             y = y + 2;
101             if (y == 15) {
102                 break;
103             }
104         }
105 
106         y = 2;
107 
108         for (Double i : budgeter.getIncome().values()) {
109             setting.add(new Text(" " + Double.toString(i)), 4, y);
110             y = y + 2;
111             if (y == 16) {
112                 setting.add(new Text("..."), 4, 15);
113                 break;
114             }
115         }
116 
117         y = 1;
118 
119         for (String n : budgeter.getExpenses().keySet()) {
120             setting.add(new Text(n), 5, y);
121             y = y + 2;
122             if (y == 15) {
123                 break;
124             }
125         }
126 
127         y = 2;
128 
129         for (Double i : budgeter.getExpenses().values()) {
130             setting.add(new Text(" " + Double.toString(i)), 5, y);
131             y = y + 2;
132             if (y == 16) {
133                 setting.add(new Text("..."), 5, 15);
134                 break;
135             }
136         }
137 
138         return setting;
139     }
140 }
