6   package gui;
7   
8   import javafx.geometry.Insets;
9   import javafx.geometry.Pos;
10  import javafx.scene.Parent;
11  import javafx.scene.control.Button;
12  import javafx.scene.control.Label;
13  import javafx.scene.control.TextField;
14  import javafx.scene.layout.GridPane;
15  import javafx.scene.paint.Color;
16  import javafx.scene.text.Text;
17  import logic.Tasklist;
18  import logic.Task;
19  
20  /**
21   * Class creates a GridPane element that lets user input tasks.
22   * @author Juho
23   */
24  public class TaskView {
25  
26      private Tasklist tasklist;
27  
28      /**
29       * Constructor that sets the attribute used to store tasks with.
30       * @param tasklist
31       */
32      public TaskView(Tasklist tasklist) {
33          this.tasklist = tasklist;
34      }
35  
36      /**
37       * Method returns a GridPane element that lets the user input task information.
38       * @return GridPane element to input info with.
39       */
40      public Parent getView() {
41          GridPane setting = new GridPane();
42  
43          setting.setAlignment(Pos.CENTER);
44          setting.setVgap(10);
45          setting.setHgap(10);
46          setting.setPadding(new Insets(10, 10, 10, 10));
47  
48          Text error = new Text("Määräsyöte ei ole numero");
49          error.setFill(Color.RED);
50          Label description = new Label("Syötä tiedossasi olevia tehtäviä ja arvioi niiden vaatimaa aikaa.");
51          Label name = new Label("Tehtävä");
52          Label motive = new Label("Motiivi");
53          Label timeInfo = new Label("Vaadittu aika (min.)");
54  
55          TextField time = new TextField();
56          TextField taskName = new TextField();
57          TextField motiveName = new TextField();
58          Button saveTask = new Button("Syötä tehtävä");
59  
60          setting.add(description, 0, 0);
61          setting.add(name, 0, 1);
62          setting.add(timeInfo, 0, 2);
63          setting.add(motive, 0, 3);
64  
65          setting.add(taskName, 1, 1);
66          setting.add(time, 1, 2);
67          setting.add(motiveName, 1, 3);
68          setting.add(saveTask, 1, 4);
69  
70          saveTask.setOnAction((event) -> {
71              try {
72                  setting.getChildren().remove(error);
73  
74                  int t = Integer.parseInt(time.getText());
75                  String n = taskName.getText();
76                  String m = motiveName.getText();
77  
78                  tasklist.addTask(new Task(n, m, t));
79  
80              } catch (Exception e) {
81                  setting.add(error, 1, 4);
82                  return;
83  
84              }
85              time.clear();
86              taskName.clear();
87              motiveName.clear();
88          });
89  
90          return setting;
91      }
92  }
