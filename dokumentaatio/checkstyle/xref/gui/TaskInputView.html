6   package gui;
7   
8   import javafx.geometry.Insets;
9   import javafx.geometry.Pos;
10  import javafx.scene.Parent;
11  import javafx.scene.control.Button;
12  import javafx.scene.control.Label;
13  import javafx.scene.control.TextField;
14  import javafx.scene.layout.GridPane;
15  import javafx.scene.paint.Color;
16  import javafx.scene.text.Text;
17  import logic.Tasklist;
18  import logic.Task;
19  import logic.User;
20  
21  /**
22   * Class creates a GridPane element that lets user input tasks.
23   *
24   * @author Juho
25   */
26  public class TaskInputView {
27  
28      private Tasklist tasklist;
29      private User user;
30  
31      /**
32       * Constructor that sets the attribute used to store tasks with.
33       *
34       * @param tasklist tasklist object given by the initializing object
35       * @param user user object given by the initializing object
36       *
37       * @see Tasklist
38       * @see User
39       */
40      public TaskInputView(Tasklist tasklist, User user) {
41          this.tasklist = tasklist;
42          this.user = user;
43      }
44  
45      /**
46       * Method returns a GridPane element that lets the user input task
47       * information.
48       *
49       * @return GridPane element to input info with.
50       */
51      public Parent getView() {
52  
53          GridPane setting = new GridPane();
54  
55          setting.setAlignment(Pos.CENTER);
56          setting.setVgap(10);
57          setting.setHgap(10);
58          setting.setPadding(new Insets(10, 10, 10, 10));
59  
60          Text numberError = new Text("Aikasyöte ei ole numero");
61          numberError.setFill(Color.RED);
62          Text negativeError = new Text("Syötä vain positiivisia arvoja");
63          negativeError.setFill(Color.RED);
64          Text error = new Text("Puutteellinen syöte");
65          error.setFill(Color.RED);
66          Label description = new Label("Syötä tiedossasi olevia tehtäviä \nja arvioi niiden vaatimaa aikaa.");
67          Label name = new Label("Tehtävä");
68          Label motive = new Label("Motiivi");
69          Label timeInfo = new Label("Vaadittu aika (min.)");
70          Label userTime = new Label("Syötä minuutteina päivittäin käytettävissä \noleva aika tehtävien tekemiseen.");
71  
72          TextField capacity = new TextField();
73          TextField time = new TextField();
74          TextField taskName = new TextField();
75          TextField motiveName = new TextField();
76  
77          Button saveTask = new Button("Syötä tehtävä");
78          Button saveCapacity = new Button("Tallenna määrä");
79          setting.add(description, 0, 0);
80          setting.add(name, 0, 1);
81          setting.add(timeInfo, 0, 2);
82          setting.add(motive, 0, 3);
83          setting.add(userTime, 0, 7);
84  
85          setting.add(taskName, 1, 1);
86          setting.add(time, 1, 2);
87          setting.add(motiveName, 1, 3);
88          setting.add(capacity, 1, 7);
89  
90          setting.add(saveTask, 1, 4);
91          setting.add(saveCapacity, 1, 8);
92  
93          saveTask.setOnAction((event) -> {
94              setting.getChildren().remove(error);
95              setting.getChildren().remove(numberError);
96              setting.getChildren().remove(negativeError);
97  
98              if (taskName.getText().isEmpty() || time.getText().isEmpty()) {         //Motive not included in check as a design choice.
99                  setting.add(error, 1, 5);
100                 return;
101             }
102 
103             if (Integer.parseInt(time.getText()) < 1) {
104                 setting.add(negativeError, 1, 5);
105                 return;
106             }
107 
108             try {
109                 int t = Integer.parseInt(time.getText());
110                 String n = taskName.getText();
111                 String m = motiveName.getText();
112 
113                 tasklist.addTask(new Task(n, m, t));
114 
115             } catch (Exception e) {
116                 setting.add(numberError, 1, 5);
117                 return;
118             }
119             time.clear();
120             taskName.clear();
121             motiveName.clear();
122         });
123 
124         saveCapacity.setOnAction((event) -> {
125             setting.getChildren().remove(error);
126             setting.getChildren().remove(numberError);
127             setting.getChildren().remove(negativeError);
128 
129             if (capacity.getText().isEmpty()) {
130                 setting.add(error, 1, 9);
131                 return;
132             }
133 
134             if (Integer.parseInt(capacity.getText()) < 1) {
135                 setting.add(negativeError, 1, 9);
136                 return;
137             }
138 
139             try {
140                 int c = Integer.parseInt(capacity.getText());
141                 user.setDailyCapacity(c);
142 
143             } catch (Exception e) {
144 
145                 setting.add(numberError, 1, 9);
146                 return;
147 
148             }
149             capacity.clear();
150         });
151 
152         return setting;
153     }
154 }
